def create_config(**kwargs):

④ 默认参数陷阱（经典面试题）
⚠️ 可变默认参数只初始化一次
# ❌ 错误写法
def add_item(item, lst=[]):  # 默认空列表只在函数定义时创建一次！
    lst.append(item)
    return lst

print(add_item(1))  # → [1]
print(add_item(2))  # → [1, 2]  ← 不是 [2]！
print(add_item(3))  # → [1, 2, 3]

# ✅ 正确写法：用 None 代替
def add_item(item, lst=None):
    if lst is None:
        lst = []
    lst.append(item)
    return lst
💡 规则：默认参数在函数定义时求值，只执行一次。所以不要用 []、{}、set() 等可变对象作为默认值，用 None 代替。

⑥ 闭包（Closure）
📌 定义：内部函数引用外部变量
def make_counter():
    count = 0                      # 外部变量
    def counter():                  # 内部函数（闭包）
        nonlocal count               # 声明要修改外部变量
        count += 1
        return count
    return counter                   # 返回内部函数

# 使用
my_counter = make_counter()         # my_counter 现在是一个闭包
print(my_counter())                 # → 1
print(my_counter())                 # → 2
print(my_counter())                 # → 3

# 独立的计数器
counter2 = make_counter()
print(counter2())                  # → 1 （独立状态！）
闭包三要素说明
① 外部函数定义一个变量
② 内部函数引用外部函数的变量
③ 返回内部函数让外部可以调用这个闭包
💡 对比 Swift 闭包

Python 闭包 ≈ Swift 中函数内部捕获外部变量：

// Swift 等效
func makeCounter() -> () -> Int {
    var count = 0
    return {                         // 闭包
        count += 1
        return count
    }
}

④ @classmethod / @staticmethod
📌 三种方法类型
class Dog:
    species = "犬科"

    def bark(self):              # ① 实例方法：需要 self
        print(f"{self.name} 汪汪!")

    @classmethod
    def get_species(cls):        # ② 类方法：需要 cls（类本身）
        return cls.species

    @staticmethod
    def is_dog(name):             # ③ 静态方法：不需要 self/cls
        return name in ["旺财", "小黑"]

⑥ __str__ / __repr__ — 打印友好
📌 等于 Swift 的 CustomStringConvertible
class Dog:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def __str__(self):        # 给 print() 和 str() 用
        return f"🐶 {self.name}（{self.age}岁）"

⑦ @dataclass — 自动生成样板代码 ⭐
📌 Day 2 最重要的知识点
from dataclasses import dataclass

# 不用 dataclass（手写一堆样板代码）
class Dog:
    def __init__(self, name: str, age: int = 0):
        self.name = name
        self.age = age
    def __repr__(self):
        return f"Dog(name={self.name}, age={self.age})"
    def __eq__(self, other):
        return self.name == other.name and self.age == other.age

# ✅ 用 dataclass（一行搞定！）
@dataclass
class Dog:
    name: str
    age: int = 0

# 自动拥有：__init__、__repr__、__eq__
dog1 = Dog("旺财", 3)
dog2 = Dog("旺财", 3)
print(dog1)          # → Dog(name='旺财', age=3)
print(dog1 == dog2)  # → True（自动比较）
@dataclass 自动生成等价于
__init__构造函数
__repr__打印友好
__eq__相等比较
💡 高级用法：field()
from dataclasses import dataclass, field
from typing import List

@dataclass
class ChatHistory:
    messages: List[str] = field(default_factory=list)  # 可变默认值用 field
    model: str = "qwen"
    temperature: float = 0.7

    def add(self, msg):
        self.messages.append(msg)

    def count(self):
        return len(self.messages)

field(default_factory=list) 解决了"默认参数陷阱"，每次创建实例都生成新列表
