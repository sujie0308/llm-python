Swift vs Python OOP 对照：

// Swift
class Dog {
    let name: String
    init(name: String) { self.name = name }
    func bark() { print("\(name) 汪汪!") }
}

# Python
class Dog:
    def __init__(self, name: str):
        self.name = name
    def bark(self):
        print(f"{self.name} 汪汪!")

# Python 用 dataclass 等价于 Swift struct
from dataclasses import dataclass

@dataclass
class Dog:
    name: str
    age: int = 0
    def bark(self):
        print(f"{self.name} 汪汪!")

必掌握知识点：

• class 与 __init__：等于 Swift 的 class 和 init

• self：等于 Swift 的 self，但必须显式写在方法第一个参数

• 实例属性 vs 类属性

• @property：等于 Swift computed property

• @classmethod / @staticmethod

• 继承：class Puppy(Dog):

• __str__ / __repr__：等于 Swift 的 CustomStringConvertible

• @dataclass：自动生成 __init__、__repr__、__eq__，**LangChain/Pydantic 大量使用** ⭐

• 没有 access control：Python 用 _private 约定（不强制）

PyTorch 的 nn.Module、LangChain 的所有 Chain、Pydantic 模型，全部是 class，必须熟练。

"""
LLMClient - 大模型客户端封装
功能：
1. 同步调用 + 流式调用
2. 多轮对话历史管理
3. 切换模型 / 温度 / system prompt
4. Token 计数与成本估算
"""
from dataclasses import dataclass, field
from typing import Optional, List, Generator
from openai import OpenAI
import os
from dotenv import load_dotenv

load_dotenv()


@dataclass
class ChatMessage:
    role: str  # "user" / "assistant" / "system"
    content: str

    def to_dict(self) -> dict:
        return {"role": self.role, "content": self.content}


@dataclass
class LLMClient:
    api_key: str = field(default_factory=lambda: os.getenv("API_KEY", ""))
    base_url: str = field(default_factory=lambda: os.getenv("API_BASE", ""))
    model: str = field(default_factory=lambda: os.getenv("MODEL", ""))
    temperature: float = 0.7
    system_prompt: str = "你是一个有帮助的AI助手。"

    def __post_init__(self):
        self._client = OpenAI(api_key=self.api_key, base_url=self.base_url)
        self._history: List[ChatMessage] = []
        self.reset()

    def reset(self) -> None:
        """重置对话历史，保留 system prompt"""
        self._history = [ChatMessage(role="system", content=self.system_prompt)]

    def set_system(self, prompt: str) -> None:
        self.system_prompt = prompt
        self.reset()

    def chat(self, user_input: str) -> str:
        """同步对话"""
        self._history.append(ChatMessage(role="user", content=user_input))
        response = self._client.chat.completions.create(
            model=self.model,
            messages=[m.to_dict() for m in self._history],
            temperature=self.temperature,
        )
        reply = response.choices[0].message.content
        self._history.append(ChatMessage(role="assistant", content=reply))
        return reply

    def stream_chat(self, user_input: str) -> Generator[str, None, None]:
        """流式对话（生成器）"""
        self._history.append(ChatMessage(role="user", content=user_input))
        stream = self._client.chat.completions.create(
            model=self.model,
            messages=[m.to_dict() for m in self._history],
            temperature=self.temperature,
            stream=True,
        )
        full_reply = ""
        for chunk in stream:
            delta = chunk.choices[0].delta.content
            if delta:
                full_reply += delta
                yield delta
        self._history.append(ChatMessage(role="assistant", content=full_reply))

    @property
    def history_length(self) -> int:
        return len(self._history)

    def show_history(self) -> None:
        for i, msg in enumerate(self._history):
            print(f"[{i}] {msg.role}: {msg.content[:80]}")

再写一个交互式 CLI： day2/cli.py

from llm_client import LLMClient

def main():
    client = LLMClient()
    print("🤖 LLM CLI 启动（quit退出 / clear清空历史 / show查看历史）\n")
    while True:
        user_input = input("你: ").strip()
        if not user_input:
            continue
        if user_input == "quit":
            break
        if user_input == "clear":
            client.reset()
            print("✅ 历史已清空\n")
            continue
        if user_input == "show":
            client.show_history()
            continue
        print("AI: ", end="", flush=True)
        for chunk in client.stream_chat(user_input):
            print(chunk, end="", flush=True)
        print("\n")

if __name__ == "__main__":
    main()

import pytest
from llm_client import LLMClient, ChatMessage


def test_chat_message_to_dict():
    msg = ChatMessage(role="user", content="hello")
    assert msg.to_dict() == {"role": "user", "content": "hello"}


def test_client_initialization():
    client = LLMClient()
    assert client.history_length == 1  # 只有 system prompt
    assert client.temperature == 0.7


def test_reset_clears_history():
    client = LLMClient()
    client._history.append(ChatMessage(role="user", content="test"))
    assert client.history_length == 2
    client.reset()
    assert client.history_length == 1


def test_set_system_resets():
    client = LLMClient()
    client.set_system("你是一只猫")
    assert client.system_prompt == "你是一只猫"
    assert client.history_length == 1


# 进阶：用 mock 测试 chat 方法（不真实调用 API）
def test_chat_with_mock(mocker):
    client = LLMClient()
    mock_response = mocker.Mock()
    mock_response.choices = [mocker.Mock()]
    mock_response.choices[0].message.content = "mocked reply"
    mocker.patch.object(client._client.chat.completions, "create", return_value=mock_response)
    reply = client.chat("hi")
    assert reply == "mocked reply"
    assert client.history_length == 3  # system + user + assistant
