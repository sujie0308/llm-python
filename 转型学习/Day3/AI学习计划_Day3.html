创建 day3/async_vs_sync.py：

"""
对比同步 vs 异步并发调用 LLM 的性能差异
"""
import time
import asyncio
import os
from openai import OpenAI, AsyncOpenAI
from dotenv import load_dotenv

load_dotenv()
API_KEY = os.getenv("API_KEY")
API_BASE = os.getenv("API_BASE")
MODEL = os.getenv("MODEL")

PROMPTS = [
    "用一句话介绍Python",
    "用一句话介绍Swift",
    "用一句话介绍Rust",
    "用一句话介绍Go",
    "用一句话介绍TypeScript",
    "用一句话介绍Java",
    "用一句话介绍C++",
    "用一句话介绍Kotlin",
    "用一句话介绍Dart",
    "用一句话介绍Ruby",
]


# ========= 同步版 =========
def sync_call(prompt: str) -> str:
    client = OpenAI(api_key=API_KEY, base_url=API_BASE)
    resp = client.chat.completions.create(
        model=MODEL,
        messages=[{"role": "user", "content": prompt}],
    )
    return resp.choices[0].message.content


def run_sync():
    start = time.time()
    results = [sync_call(p) for p in PROMPTS]
    elapsed = time.time() - start
    print(f"\n🐢 同步耗时: {elapsed:.2f}s")
    return elapsed


# ========= 异步版 =========
async def async_call(client: AsyncOpenAI, prompt: str) -> str:
    resp = await client.chat.completions.create(
        model=MODEL,
        messages=[{"role": "user", "content": prompt}],
    )
    return resp.choices[0].message.content


async def run_async():
    client = AsyncOpenAI(api_key=API_KEY, base_url=API_BASE)
    start = time.time()
    results = await asyncio.gather(*[async_call(client, p) for p in PROMPTS])
    elapsed = time.time() - start
    print(f"\n🚀 异步耗时: {elapsed:.2f}s")
    return elapsed


if __name__ == "__main__":
    print(f"测试 {len(PROMPTS)} 个 LLM 请求...")
    t_sync = run_sync()
    t_async = asyncio.run(run_async())
    print(f"\n✨ 异步比同步快 {t_sync / t_async:.1f} 倍")

运行：python day3/async_vs_sync.py

用数据说话，建立对异步性能优势的肌肉记忆。
看到异步比同步快至少 3 倍（一般 5-10 倍）。

创建 day3/async_llm_client.py：

from dataclasses import dataclass, field
from typing import List, Optional, AsyncGenerator, Literal
from openai import AsyncOpenAI
import os
import time
import functools
from dotenv import load_dotenv

load_dotenv()

Role = Literal["user", "assistant", "system"]


def async_timer(func):
    """异步函数计时装饰器"""
    @functools.wraps(func)
    async def wrapper(*args, **kwargs):
        start = time.time()
        result = await func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"⏱  {func.__name__} 耗时 {elapsed:.2f}s")
        return result
    return wrapper


@dataclass
class ChatMessage:
    role: Role
    content: str

    def to_dict(self) -> dict[str, str]:
        return {"role": self.role, "content": self.content}


@dataclass
class AsyncLLMClient:
    api_key: str = field(default_factory=lambda: os.getenv("API_KEY", ""))
    base_url: str = field(default_factory=lambda: os.getenv("API_BASE", ""))
    model: str = field(default_factory=lambda: os.getenv("MODEL", ""))
    temperature: float = 0.7
    system_prompt: str = "你是一个有帮助的AI助手。"

    def __post_init__(self) -> None:
        self._client = AsyncOpenAI(api_key=self.api_key, base_url=self.base_url)
        self._history: List[ChatMessage] = []
        self.reset()

    def reset(self) -> None:
        self._history = [ChatMessage(role="system", content=self.system_prompt)]

    @async_timer
    async def chat(self, user_input: str) -> str:
        self._history.append(ChatMessage(role="user", content=user_input))
        resp = await self._client.chat.completions.create(
            model=self.model,
            messages=[m.to_dict() for m in self._history],
            temperature=self.temperature,
        )
        reply = resp.choices[0].message.content or ""
        self._history.append(ChatMessage(role="assistant", content=reply))
        return reply

    async def stream_chat(self, user_input: str) -> AsyncGenerator[str, None]:
        self._history.append(ChatMessage(role="user", content=user_input))
        stream = await self._client.chat.completions.create(
            model=self.model,
            messages=[m.to_dict() for m in self._history],
            stream=True,
            temperature=self.temperature,
        )
        full = ""
        async for chunk in stream:
            delta = chunk.choices[0].delta.content
            if delta:
                full += delta
                yield delta
        self._history.append(ChatMessage(role="assistant", content=full))


# 异步主程序示例
async def main():
    client = AsyncLLMClient()
    reply = await client.chat("你好")
    print("回复:", reply)
    print("\n流式输出:")
    async for chunk in client.stream_chat("用一句话介绍Python"):
        print(chunk, end="", flush=True)
    print()


if __name__ == "__main__":
    import asyncio
    asyncio.run(main())
